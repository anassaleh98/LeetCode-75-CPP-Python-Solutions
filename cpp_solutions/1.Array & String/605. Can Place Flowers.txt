class Solution {
public:
    bool canPlaceFlowers(vector<int>& arr, int n) {
        int s = arr.size() , c=0;
        if(s==1 && n==1 && arr[0]==0) c++;
        if (s>1 && arr[0] == 0 && arr[1] ==0) {arr[0]=1, c++;}
        if (s>1 && arr[s-1] == 0 && arr[s-2] ==0) {arr[s-1]=1, c++;}
        for(int i =1; i<s-1; i++){
            if(arr[i]==0 && arr[i-1]==0 && arr[i+1]==0) {arr[i]=1, c++;}
        }
    if(c>=n) return 1;
    else return 0;
    }
};
