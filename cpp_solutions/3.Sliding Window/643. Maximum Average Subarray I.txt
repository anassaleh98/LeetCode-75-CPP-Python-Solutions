class Solution {
public:
    double findMaxAverage(vector<int>& arr, int k) {

        for(int i=1; i<arr.size(); i++){arr[i]+=arr[i-1];} 

        int l=1, r=k; 
        double mx = (double)arr[k - 1] / k;
        while(r <arr.size()){
            double avg = (double)(arr[r]-arr[l-1])/k;
            mx = max(avg, mx);
            r++, l++;
        }
        return round(mx * 100000) / 100000;
    }
};
