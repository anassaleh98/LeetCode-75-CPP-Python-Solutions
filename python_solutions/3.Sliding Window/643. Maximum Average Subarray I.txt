class Solution:
    def findMaxAverage(self, arr: List[int], k: int) -> float:

        for i in range(1, len(arr)):
            arr[i] += arr[i-1] 

        l=1
        r=k 
        mx = arr[k - 1] / k
        while(r < len(arr)):
            avg = (arr[r]-arr[l-1])/k
            mx = max(avg, mx)
            r+= 1
            l+= 1
        
        return round(mx ,5)
